"""Credentials I/O and OAuth refresh primitives for claude-quota-exporter.

Pure-function module. State and locking live in Fetcher.
"""

from __future__ import annotations

import json
import logging
import time
from dataclasses import dataclass
from datetime import datetime
from pathlib import Path

logger = logging.getLogger(__name__)


@dataclass(frozen=True)
class Credentials:
    """Snapshot of OAuth credentials in memory.

    `expires_at` and `issued_at` are Unix epoch seconds (float).
    Disk format uses milliseconds — converted at the boundary.
    """

    access_token: str
    refresh_token: str
    expires_at: float
    issued_at: float


def _parse_expires_at(value: object) -> float | None:
    """Accept ms-int (Claude Code) or ISO-8601 string. Return epoch seconds."""
    if isinstance(value, (int, float)) and not isinstance(value, bool):
        return float(value) / 1000.0
    if isinstance(value, str) and value:
        try:
            return datetime.fromisoformat(value).timestamp()
        except (ValueError, TypeError):  # fmt: skip
            return None
    return None


def read_credentials(path: Path) -> tuple[Credentials, dict] | None:
    """Parse credentials from disk.

    Returns (Credentials, raw_dict) on success. The raw dict is the full
    parsed JSON — used at write-back time to preserve unknown sibling
    fields. Returns None for any failure (missing file, malformed JSON,
    missing/empty required fields, unparseable expiresAt). Never raises.
    """
    try:
        text = path.read_text(encoding="utf-8")
    except (OSError,):  # fmt: skip
        return None
    try:
        data = json.loads(text)
    except json.JSONDecodeError:
        return None
    if not isinstance(data, dict):
        return None
    section = data.get("claudeAiOauth")
    if not isinstance(section, dict):
        return None
    access = section.get("accessToken")
    refresh = section.get("refreshToken")
    if not isinstance(access, str) or not access:
        return None
    if not isinstance(refresh, str) or not refresh:
        return None
    expires_at = _parse_expires_at(section.get("expiresAt"))
    if expires_at is None:
        return None
    creds = Credentials(
        access_token=access,
        refresh_token=refresh,
        expires_at=expires_at,
        issued_at=time.time(),
    )
    return creds, data
```

import urllib.error
import urllib.request

DEFAULT_EXPIRES_IN_SECONDS = 3600


def refresh(
    *,
    refresh_token: str,
    urls: list[str],
    client_id: str,
    user_agent: str,
    timeout_seconds: int,
) -> Credentials | None:
    """Exchange `refresh_token` for a new credential set.

    Tries each URL in order. Returns a Credentials snapshot on the first
    success. Returns None when every URL fails (network error, non-2xx,
    non-JSON body, or 2xx with no `access_token`).
    """
    body = json.dumps(
        {
            "grant_type": "refresh_token",
            "refresh_token": refresh_token,
            "client_id": client_id,
        }
    ).encode("utf-8")
    headers = {
        "Content-Type": "application/json",
        "User-Agent": user_agent,
    }
    for url in urls:
        req = urllib.request.Request(
            url, data=body, headers=headers, method="POST"
        )
        try:
            with urllib.request.urlopen(req, timeout=timeout_seconds) as resp:
                payload = json.loads(resp.read())
        except (
            urllib.error.URLError,
            TimeoutError,
            json.JSONDecodeError,
        ):  # fmt: skip
            logger.warning("oauth refresh failed url=%s error=unknown", url)
            continue
        if not isinstance(payload, dict):
            logger.warning("oauth refresh failed url=%s error=non_dict_body", url)
            continue
        access = payload.get("access_token")
        if not isinstance(access, str) or not access:
            logger.warning(
                "oauth refresh failed url=%s error=missing_access_token", url
            )
            continue
        new_refresh = payload.get("refresh_token")
        if not isinstance(new_refresh, str) or not new_refresh:
            new_refresh = refresh_token
        expires_in_raw = payload.get("expires_in", DEFAULT_EXPIRES_IN_SECONDS)
        try:
            expires_in = float(expires_in_raw)
        except (TypeError, ValueError):  # fmt: skip
            expires_in = float(DEFAULT_EXPIRES_IN_SECONDS)
        now = time.time()
        logger.info("oauth refresh succeeded url=%s", url)
        return Credentials(
            access_token=access,
            refresh_token=new_refresh,
            expires_at=now + expires_in,
            issued_at=now,
        )
    return None
```

def _extract_oauth_error(exc: urllib.error.HTTPError) -> str:
    """Pull the RFC-6749 `error` field from an HTTPError body if present."""
    try:
        raw = exc.read()
    except Exception:  # noqa: BLE001 — body read is best-effort
        return "unknown"
    try:
        body = json.loads(raw)
    except (json.JSONDecodeError, TypeError):  # fmt: skip
        return "unknown"
    if isinstance(body, dict):
        code = body.get("error")
        if isinstance(code, str) and code:
            return code
    return "unknown"

import os


def is_writable(path: Path) -> bool:
    """Probe whether we could later atomically rewrite `path`.

    Creates and deletes a sibling temp file. Returns True if both
    succeed. The probe does not modify or read the target file.
    """
    probe = path.with_suffix(path.suffix + ".write_probe")
    try:
        probe.write_bytes(b"")
        probe.unlink()
        return True
    except OSError:
        return False


def write_back(path: Path, creds: Credentials, raw_dict: dict) -> bool:
    """Atomically write the credentials back to disk.

    Updates only the three fields we own
    (accessToken/refreshToken/expiresAt) inside claudeAiOauth; preserves
    every sibling field. Writes to a temp file in the same directory,
    fsyncs, os.replaces over the original, and re-applies the source
    file mode. Never raises. Returns True on success, False on any
    failure.
    """
    section = raw_dict.setdefault("claudeAiOauth", {})
    if not isinstance(section, dict):
        return False
    section["accessToken"] = creds.access_token
    section["refreshToken"] = creds.refresh_token
    section["expiresAt"] = int(creds.expires_at * 1000)
    try:
        src_mode = path.stat().st_mode & 0o777
    except OSError:
        return False
    tmp = path.with_suffix(path.suffix + ".tmp")
    try:
        with tmp.open("w", encoding="utf-8") as f:
            json.dump(raw_dict, f, indent=2)
            f.flush()
            os.fsync(f.fileno())
        os.replace(tmp, path)
    except OSError:
        try:
            tmp.unlink()
        except OSError:
            pass
        return False
    try:
        os.chmod(path, src_mode)
    except OSError:
        pass  # Windows; non-fatal
    return True
```

            if self._creds is None:
                logger.debug(
                    "no credentials loaded at %s; skip fetch",
                    self._credentials_path,
                )
                return False
            # Proactive refresh: token within skew window.
            if (
                self._creds.expires_at - now
                < self._refresh_skew_seconds
            ):
                self._do_refresh()
            self._last_attempt_at = now
            assert self._creds is not None  # refresh failure preserves old creds
            raw = fetch_from_api(
                token=self._creds.access_token,
                timeout_seconds=self._timeout_seconds,
            )
            if raw is None:
                self._consecutive_failures += 1
                self._failure_count += 1
                logger.warning(
                    "limits fetch failed attempts=%d backoff=%.0fs",
                    self._consecutive_failures,
                    self._backoff_seconds(),
                )
                return False
            self._cached_raw = raw
            self._last_success_at = time.time()
            self._consecutive_failures = 0
            self._success_count += 1
            logger.info("limits fetched and cached")
            return True

    def _do_refresh(self) -> bool:
        """Run a refresh, persist on success. Caller holds _lock."""
        assert self._creds is not None
        new_creds = auth.refresh(
            refresh_token=self._creds.refresh_token,
            urls=self._refresh_urls,
            client_id=self._oauth_client_id,
            user_agent=self._refresh_user_agent,
            timeout_seconds=self._timeout_seconds,
        )
        if new_creds is None:
            self._refresh_failure_count += 1
            return False
        self._creds = new_creds
        self._last_refresh_at = time.time()
        self._refresh_success_count += 1
        if self._credentials_writable:
            written = auth.write_back(
                self._credentials_path, new_creds, self._raw_creds_blob
            )
            if not written:
                self._credentials_writable = False
                logger.warning(
                    "credentials write-back failed; running in-memory only"
                )
        return True
```

After the existing `## Endpoints` section, insert:

````markdown
## Authentication and token refresh

The exporter consumes an OAuth credentials file in the
[Claude Code shape](https://code.claude.com/docs/en/authentication).
The credentials file (`credentials_path` in config) must contain:

```json
{
  "claudeAiOauth": {
    "accessToken": "sk-ant-oat01-...",
    "refreshToken": "sk-ant-ort01-...",
    "expiresAt": 1779580800694
  }
}
```

Other fields (`scopes`, `subscriptionType`, etc.) are preserved
verbatim on write-back.

The exporter refreshes proactively when the access token is within
`refresh_skew_seconds` (default 300) of expiry, and reactively on a
usage-endpoint 401 only when the local expiry math agrees the token
is past its skew window AND no refresh has fired within the last
skew window. Refresh tokens rotate on every successful refresh; the
new pair is written back atomically.

**Writability contract.** `credentials_path` must be writable for
the exporter to survive token rotation across process restarts. If
the file is read-only the exporter still runs — refresh works
in-memory only and the rotated pair is lost when the Pod restarts.
`claude_quota_credentials_writable` reports the writability state.

**Refresh-token expiry is not surfaced.** Anthropic's refresh
endpoint does not return refresh-token expiry. The exporter exposes
`claude_quota_refresh_token_issued_at_seconds` (when the current
refresh token was first observed locally) as the closest proxy.
````
